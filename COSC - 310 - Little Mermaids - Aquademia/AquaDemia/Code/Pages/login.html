function displayAlert(message) { alert(message); } function fetchAlert() { var xhttp = new XMLHttpRequest(); xhttp.onreadystatechange = function() { if (this.readyState == 4 && this.status == 200) { displayAlert(this.responseText); } }; xhttp.open("GET", "loginPage.php", true); xhttp.send(); } // Fetch alert when the page loads window.onload = function() { fetchAlert(); };
